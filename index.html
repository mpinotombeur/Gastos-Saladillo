function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName('Registro de gastos (1)'); const values = sheet.getDataRange().getValues(); const VALID_CATS = ['Vivienda','Comida','Movilidad','Recreación','Malta']; const MONTHS = { 1:'Enero',2:'Febrero',3:'Marzo',4:'Abril',5:'Mayo',6:'Junio', 7:'Julio',8:'Agosto',9:'Septiembre',10:'Octubre',11:'Noviembre',12:'Diciembre' }; const result = []; let id = 1; for (let i = 1; i < values.length; i++) { const row = values[i]; const fecha = row[0]; const cat = String(row[1]||'').trim(); const sub = String(row[2]||'').trim(); const monto = row[3]; const pago = String(row[4]||'').trim(); const quien = String(row[5]||'').trim(); const obs = String(row[6]||'').trim(); // Skip rows without valid category or monto if (!VALID_CATS.includes(cat)) continue; if (!monto || isNaN(parseFloat(monto))) continue; if (!fecha) continue; // Parse fecha let fechaStr = ''; let mes = ''; try { let d; if (fecha instanceof Date) { d = fecha; } else { const parts = String(fecha).split('/'); if (parts.length === 3) { d = new Date(parts[2], parts[1]-1, parts[0]); } else { d = new Date(fecha); } } if (!isNaN(d)) { const y = d.getFullYear(); const m = String(d.getMonth()+1).padStart(2,'0'); const day = String(d.getDate()).padStart(2,'0'); fechaStr = y + '-' + m + '-' + day; mes = MONTHS[d.getMonth()+1] || ''; } } catch(err) { continue; } if (!fechaStr || !mes) continue; const montoNum = parseFloat(String(monto).replace(/[^0-9.,]/g,'').replace(',','.')); if (!montoNum || isNaN(montoNum)) continue; result.push({ id: id++, Fecha: fechaStr, Mes: mes, 'Categoría': cat, 'Subcategoría': sub, Monto: Math.round(montoNum), 'Forma de pago': pago, 'Quién pagó': quien, Observaciones: obs }); } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); }
